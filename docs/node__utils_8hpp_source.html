30 inline array_mml<size_t> compute_pool_output_shape(
31 const array_mml<size_t>& input_shape,
32 const std::string& auto_pad,
33 const int ceil_mode,
34 const std::vector<int>& dilations,
35 const std::vector<int>& kernel_shape,
36 const std::vector<int>& pads,
37 const std::vector<int>& strides
38 ) {
39 size_t spacial_rank = kernel_shape.size();
40 std::vector<size_t> output_shape_vector = { input_shape[0], input_shape[1] };
41
42 for (size_t i = 0; i < spacial_rank; ++i) {
43 int input_dim = input_shape[i + 2];
44 int kernel = kernel_shape[i];
45 int stride = strides[i];
46 int dilation = dilations[i];
47
48 int effective_kernel = (kernel - 1) * dilation + 1;
49
50 size_t out_dim;
51 if (auto_pad == "SAME_UPPER" || auto_pad == "SAME_LOWER") {
52 if (ceil_mode) {
53 out_dim = std::ceil(static_cast<float>(input_dim) / stride);
54 } else {
55 out_dim = std::floor(static_cast<float>((input_dim - 1) / stride)) + 1;
56 }
57 } else if (auto_pad == "VALID") {
58 if (ceil_mode) {
59 out_dim = std::ceil(static_cast<float>(input_dim - effective_kernel + 1) / stride);
60 } else {
61 out_dim = std::floor(static_cast<float>(input_dim - effective_kernel) / stride) + 1;
62 }
63 } else {
64 int total_pad = pads[i] + pads[i + spacial_rank];
65 if (ceil_mode) {
66 out_dim = std::ceil(static_cast<float>(input_dim + total_pad - effective_kernel) / stride + 1);
67 } else {
68 out_dim = std::floor(static_cast<float>(input_dim + total_pad - effective_kernel) / stride + 1);
69 }
70 }
71
72 output_shape_vector.push_back(out_dim);
73 }
74 return array_mml<size_t>(output_shape_vector);
