43 if (init.contains("rawData")) {

46 } else {
47 std::string fieldName;
48
49 if constexpr (std::is_same_v<T, float>)
50 fieldName = "floatData";
51 else if constexpr (std::is_same_v<T, double>)
52 fieldName = "doubleData";
53 else if constexpr (std::is_same_v<T, int64_t>)
54 fieldName = "int64Data";
55 else if constexpr (std::is_same_v<T, int32_t>)
56 fieldName = "int32Data";
57 else if constexpr (std::is_same_v<T, uint64_t>)
58 fieldName = "uint64Data";
59 else if constexpr (std::is_same_v<T, uint32_t>)
60 fieldName = "uint32Data";
61 else if constexpr (std::is_same_v<T, uint16_t>)
62 fieldName = "uint16Data";
63 else if constexpr (std::is_same_v<T, int16_t>)
64 fieldName = "int16Data";
65 else if constexpr (std::is_same_v<T, uint8_t>)
66 fieldName = "uint8Data";
67 else if constexpr (std::is_same_v<T, int8_t>)
68 fieldName = "int8Data";
69 else if constexpr (std::is_same_v<T, bool>)
70 fieldName = "boolData";
71 else
72 fieldName = "unknownData";
73
74 if (init.contains(fieldName)) {
75 std::vector<T> data;
76 for (const auto &el : init[fieldName]) {
77 if (el.is_number()) {
78 data.push_back(el.get<T>());
79 } else if (el.is_string()) {
80 if constexpr (std::is_same_v<T, bool>) {
81 // Handle boolean strings like "true", "false", "1", "0"
82 std::string value = el.get<std::string>();
83 std::transform(value.begin(), value.end(), value.begin(),
84 [](unsigned char c) { return std::tolower(c); });
85
86 if (value == "true" || value == "1" || value == "yes") {
87 data.push_back(true);
88 } else if (value == "false" || value == "0" || value == "no") {
89 data.push_back(false);
90 } else {
91 throw std::runtime_error("Invalid boolean std::string: " + value);
92 }
93 } else if constexpr (std::is_same_v<T, int64_t>) {
94 data.push_back(static_cast<T>(std::stoll(el.get<std::string>())));
95 } else if constexpr (std::is_same_v<T, uint64_t>) {
96 data.push_back(static_cast<T>(std::stoull(el.get<std::string>())));
97 } else if constexpr (std::is_integral_v<T>) {
98 data.push_back(static_cast<T>(std::stoi(el.get<std::string>())));
99 } else if constexpr (std::is_floating_point_v<T>) {
100 data.push_back(static_cast<T>(std::stod(el.get<std::string>())));
101 } else {
102 throw std::runtime_error(
103 "Invalid std::string conversion for type: " +
104 init["name"].get<std::string>());
105 }
106 } else if (el.is_boolean()) {
107 data.push_back(static_cast<T>(el.get<bool>()));
108 } else {
109 throw std::runtime_error("Invalid data type in tensor: " +
110 init["name"].get<std::string>());
111 }
112 }
113 array_mml<T> dataArray(data);
114 return TensorFactory::create_tensor<T>(shapeArray, dataArray);
115 } else {
116 throw std::runtime_error("No data field found for tensor: " +
117 init["name"].get<std::string>());
118 }
